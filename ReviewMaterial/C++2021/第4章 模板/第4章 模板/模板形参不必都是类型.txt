#include<iostream>
using namespace std;

template<typename T, size_t N>  // template<typename T, int N>
void array_init(T (&parm)[N]){
	for(size_t i=0;i<N;i++)
		parm[i] = i;
}

int main(){
	int int_array[10];
	double double_array[20];
	
	array_init(int_array);
	array_init(double_array);
	
	for(int i=0;i<10;i++)
		cout<<int_array[i]<<" ";
	cout<<endl;
	
	for(int i=0;i<20;i++)
		cout<<double_array[i]<<" ";
	cout<<endl;
		
		
	return 0;
}
