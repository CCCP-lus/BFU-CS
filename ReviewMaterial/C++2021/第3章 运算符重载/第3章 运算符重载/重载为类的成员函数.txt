#include<iostream>
#include<string>
#include<cstdlib>
using namespace std;


class MyString{
private:
	char s[50];
public:
	MyString(const char * s);
	bool operator!();
};

MyString::MyString(const char * s){ 
	strcpy_s(this->s, s);
}

bool MyString::operator!(){

	else return false;
}
